# Estado general
docker compose ps
docker ps --format "table {{.Names}}\t{{.Image}}\t{{.Status}}\t{{.Ports}}"

# Logs por servicio
docker compose logs --tail=200 nginx
docker compose logs --tail=200 api
docker compose logs --tail=200 db

# Ver health real (si existe)
docker inspect reverse-proxy --format '{{json .State.Health}}' | jq
docker inspect node-api --format '{{json .State.Health}}' | jq
docker inspect postgres-db --format '{{json .State.Health}}' | jq

# Ver redes y DNS interno (servicios)
docker network ls
docker network inspect devops-stack_devops-net | jq '.[0].Containers'

# Probar endpoints
curl -i http://127.0.0.1:8081/
curl -i http://127.0.0.1:8081/health
curl -i http://127.0.0.1:8081/hit

# Reinicios controlados
docker compose restart nginx
docker compose restart api
docker compose restart db

# Bajar stack
docker compose down
docker compose down -v
